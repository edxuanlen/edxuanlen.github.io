### 递归解全排列
思想如下
{% asset_img 2.png %}
```C++
#include<bits/stdc++.h>
using namespace std;
char a[3]={'a','b','c'};
int resove(int n){
        if(n==3){
            for(int i=0;i<3;i++)
            	cout<<a[i]; 
            cout<<endl;
            return 0;
        }
        for(int i=n;i<3;i++){
            swap(a[n],a[i]);
            resove(n+1);
            swap(a[n],a[i]);
        }
}
int main(){
    resove(0);
    return 0;
}
```

```c++
#include<bits/stdc++.h>
using namespace std;
char a[3]={'a','b','a'};
bool same(char *a,int n,int i){
	for(int j=n;j<i;j++)
		if(a[j]==a[i])
			return 1;
	return 0;
}
int resove(int n){
        if(n==3){
            for(int i=0;i<3;i++)
            	cout<<a[i]; 
            cout<<endl;
            return 0;
        }
        for(int i=n;i<3;i++){
        	if(!same(a,n,i)){
        		swap(a[n],a[i]);
	            resove(n+1);
	            swap(a[n],a[i]);
			}
        }
}
int main(){
    resove(0);
    return 0;
}
```
